    • 5=='5'   true
    • 5==='5'   false

    • 语法
    function 变量名(参数1, 参数2){
    函数体;
}
    • 实例
    //定义一个求和 的函数
function addNum(n1, n2){
    return n1 + n2;
}

  • 创建对象

    • 三种方式
    //方式一
var student = new Object();    //生成空对象
student.name = 'aaa';   //设置对象属性
student.age = 10;
student.showName = function(){     //成员方法,只能采用匿名函数
    console.log(this.name);
}
student.showName();

//方式二
var student = {
    name: 'bbb',    //属性,注意末尾使用的是逗号不是分号
    age: 10,
    showName: function(){     //成员方法,只能采用匿名函数
        console.log(this.name);
    }
};   //注意：此处要加分号
console.log(student.name);   //使用对象的属性
student.showName();    //调用对象的方法

//方式三
function createStudent(name, age){    //作为一个生成对象的函数
    var o = new Object();    //为对象的属性赋值
    o.name = name;
    o.age = age;
    o.showName = function(){     //只可采用匿名函数
        console.log(this.name);
    }
    return o;   //作为函数的结果返回出去
}
stu1 = createStudent('ccc', 20);
stu1.showName();

    注意:方法三是最常使用的一种创建对象的方法，是利用工厂模式的设计思想。这样可避免编写重复的代码

arr[0] = 1;   //再为数组对象赋值
arr[2] = 2;

//方式二
var arr = new Array(1, 2, 3);

//方式三
var arr = [1, 2, 3];
    • 数组方法和属性
    var len = arr.length;     //获取数组的元素个数

  • for循环

    var week = ['Monday', 'Tuesday', 'Wednesday', 'Thursday','Friday','Saturday','Weekend'];

//传统遍历
for(var i=0;i<week.length;i++){
console.log(week[i]);
}

//for-in，专注下标
for(var c in week){
console.log(week[c]);
}

//for-of，专注元素
for(var c of week){
console.log(c);
}

//高级遍历
week.forEach(c => console.log(c));
var other = week.map(c => c + 'X');//map不仅遍历，还返回另一个数组
console.log(other);
